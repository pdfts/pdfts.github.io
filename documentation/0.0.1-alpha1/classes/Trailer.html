  /**
   *
   *
   * @param {number} size
   * @param {number} startXref
   * @returns {string[]}
   * @memberof Trailer
   */
  compile(size: number, startXref: number): string[] {
    return [
      'trailer',
      '<<',
      '/ID [',
      `  <${this.ID}>`,
      `  <${this.ID}>`,
      ']',
      '/Root 1 0 R',
      '/Size ' + size,
      '>>',
      'startxref',
      startXref.toString()
    ];
  }
}
